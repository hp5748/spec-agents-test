## 2. 容器化部署

### 2.1 Dockerfile最佳实践

```dockerfile
# Dockerfile - 生产级Agent服务
FROM python:3.11-slim

# 元数据
LABEL maintainer="your-team"
LABEL version="1.0"
LABEL description="AI Agent Service"

# 设置环境变量
ENV PYTHONDONTWRITEBYTECODE=1 \
    PYTHONUNBUFFERED=1 \
    PIP_NO_CACHE_DIR=1 \
    PIP_DISABLE_PIP_VERSION_CHECK=1

# 创建非root用户
RUN groupadd -r agent && useradd -r -g agent agent

# 工作目录
WORKDIR /app

# 安装系统依赖
RUN apt-get update && apt-get install -y --no-install-recommends \
    build-essential \
    curl \
    && rm -rf /var/lib/apt/lists/*

# 复制依赖文件
COPY requirements.txt .

# 安装Python依赖
RUN pip install --no-cache-dir -r requirements.txt

# 复制代码
COPY --chown=agent:agent . .

# 切换用户
USER agent

# 健康检查
HEALTHCHECK --interval=30s --timeout=10s --start-period=5s --retries=3 \
    CMD curl -f http://localhost:8000/health || exit 1

# 暴露端口
EXPOSE 8000

# 启动命令
CMD ["uvicorn", "main:app", "--host", "0.0.0.0", "--port", "8000"]
```

### 5.1 Prometheus指标

```python
# monitoring/metrics.py
from prometheus_client import Counter, Histogram, Gauge
import time

# 定义指标
REQUEST_COUNT = Counter(
    'agent_requests_total',
    'Total number of agent requests',
    ['method', 'endpoint', 'status']
)

REQUEST_LATENCY = Histogram(
    'agent_request_latency_seconds',
    'Request latency in seconds',
    ['method', 'endpoint']
)

LLM_CALLS = Counter(
    'agent_llm_calls_total',
    'Total LLM API calls',
    ['model', 'status']
)

LLM_TOKENS = Counter(
    'agent_llm_tokens_total',
    'Total LLM tokens used',
    ['model', 'type']  # type: input/output
)

TOOL_CALLS = Counter(
    'agent_tool_calls_total',
    'Total tool calls',
    ['tool_name', 'status']
)

ACTIVE_SESSIONS = Gauge(
    'agent_active_sessions',
    'Number of active sessions'
)

# 中间件
class MetricsMiddleware:
    def __init__(self, app):
        self.app = app
    
    async def __call__(self, scope, receive, send):
        if scope["type"] == "http":
            method = scope["method"]
            path = scope["path"]
            
            start_time = time.time()
            
            # 调用下一个处理器
            await self.app(scope, receive, send)
            
            # 记录指标
            duration = time.time() - start_time
            REQUEST_LATENCY.labels(method=method, endpoint=path).observe(duration)
        
        else:
            await self.app(scope, receive, send)

# 在Agent中使用
def track_llm_call(model: str, input_tokens: int, output_tokens: int, success: bool):
    status = "success" if success else "error"
    LLM_CALLS.labels(model=model, status=status).inc()
    LLM_TOKENS.labels(model=model, type="input").inc(input_tokens)
    LLM_TOKENS.labels(model=model, type="output").inc(output_tokens)

def track_tool_call(tool_name: str, success: bool):
    status = "success" if success else "error"
    TOOL_CALLS.labels(tool_name=tool_name, status=status).inc()
```

### 6.1 结构化日志

```python
# logging_config.py
import logging
import json
from datetime import datetime

class StructuredLogger:
    """结构化日志器"""
    
    def __init__(self, name: str):
        self.logger = logging.getLogger(name)
        self.logger.setLevel(logging.INFO)
        
        # JSON格式handler
        handler = logging.StreamHandler()
        handler.setFormatter(JSONFormatter())
        self.logger.addHandler(handler)
    
    def log(self, level: str, message: str, **kwargs):
        """记录日志"""
        log_data = {
            "timestamp": datetime.utcnow().isoformat(),
            "level": level,
            "message": message,
            **kwargs
        }
        
        getattr(self.logger, level)(json.dumps(log_data))

class JSONFormatter(logging.Formatter):
    def format(self, record):
        log_data = {
            "timestamp": datetime.utcnow().isoformat(),
            "level": record.levelname,
            "logger": record.name,
            "message": record.getMessage(),
            "module": record.module,
            "function": record.funcName,
            "line": record.lineno
        }
        
        if hasattr(record, 'extra_data'):
            log_data.update(record.extra_data)
        
        return json.dumps(log_data)

# 使用示例
logger = StructuredLogger("agent")

def process_query(query: str, user_id: str):
    logger.log(
        "info",
        "Processing query",
        query=query,
        user_id=user_id,
        action="process_query"
    )
    
    try:
        result = agent.run(query)
        
        logger.log(
            "info",
            "Query processed successfully",
            user_id=user_id,
            result_length=len(result)
        )
        
        return result
    
    except Exception as e:
        logger.log(
            "error",
            "Query processing failed",
            user_id=user_id,
            error=str(e),
            error_type=type(e).__name__
        )
        raise
```

def verify_signature(request: Request, secret: str):
    """验证请求签名"""
    timestamp = request.headers.get("X-Timestamp")
    signature = request.headers.get("X-Signature")
    
    # 检查时间戳（防重放）
    if abs(time.time() - float(timestamp)) > 300:  # 5分钟有效期
        raise HTTPException(status_code=401, detail="Request expired")
    
    # 计算签名
    body = await request.body()
    expected_sig = hashlib.sha256(
        f"{timestamp}{body}{secret}".encode()
    ).hexdigest()
    
    if signature != expected_sig:
        raise HTTPException(status_code=401, detail="Invalid signature")

    def __init__(self):
        key = os.environ.get("ENCRYPTION_KEY")
        self.cipher = Fernet(key)
    
    def encrypt(self, data: str) -> str:
        """加密敏感数据"""
        return self.cipher.encrypt(data.encode()).decode()
    
    def decrypt(self, encrypted_data: str) -> str:
        """解密敏感数据"""
        return self.cipher.decrypt(encrypted_data.encode()).decode()

class SensitiveDataFilter:
    """敏感信息过滤器"""
    
    SENSITIVE_PATTERNS = [
        (r'\b\d{16,19}\b', '[CARD_NUMBER]'),  # 银行卡
        (r'\b\d{17,18}\b', '[ID_NUMBER]'),    # 身份证
        (r'\b[\w\.-]+@[\w\.-]+\.\w+\b', '[EMAIL]'),  # 邮箱
        (r'\b1[3-9]\d{9}\b', '[PHONE]'),      # 手机号
    ]
    
    @classmethod
    def mask(cls, text: str) -> str:
        """脱敏处理"""
        import re
        for pattern, replacement in cls.SENSITIVE_PATTERNS:
            text = re.sub(pattern, replacement, text)
        return text

# recovery/disaster_recovery.py
class DisasterRecovery:
    """灾备恢复"""
    
    def __init__(self):
        self.backup_location = "s3://my-bucket/agent-backup"
        self.standby_region = "us-west-2"
    
    async def failover(self):
        """故障转移"""
        # 1. 切换DNS到备用区域
        await self.switch_dns()
        
        # 2. 激活备用服务
        await self.activate_standby()
        
        # 3. 恢复最新数据
        await self.restore_data()
        
        # 4. 验证服务
        await self.verify_service()
    
    async def restore_data(self):
        """数据恢复"""
        # 下载最新备份
        latest_backup = await self.get_latest_backup()
        
        # 恢复数据库
        await self.restore_database(latest_backup["db"])
        
        # 恢复向量数据库
        await self.restore_vector_db(latest_backup["chroma"])
    
    async def health_check(self):
        """健康检查"""
        checks = {
            "database": await self.check_database(),
            "redis": await self.check_redis(),
            "vector_db": await self.check_vector_db(),
            "llm_api": await self.check_llm_api()
        }
        
        return all(checks.values()), checks
